  }

  // Progress bar
  updateProgress(cid,sess.userId,course);
  document.getElementById('topTitle').textContent=course.title;
  renderSidebar(cid,chIdx,lIdx,sess.userId);
}

function updateProgress(cid,uid,course){
  const total=course.chapters.reduce((s,ch)=>s+(ch.lessons?.length||0),0);
  const done=getLessonProg(uid,cid).length;
  const pct=total?Math.round((done/total)*100):0;
  const fill=document.getElementById('progFill');
  const txt=document.getElementById('progTxt');
  if(fill)fill.style.width=pct+'%';
  if(txt)txt.textContent=pct+'% complété';
}
